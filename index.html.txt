                <div class="space-y-4">
                    <div>
                        <span class="text-xs font-bold text-slate-400 uppercase tracking-wider">Cumulative Growth Curve: \(f(x)\)</span>
                        <div class="p-3 bg-slate-50 rounded-xl mt-1 text-center font-mono border border-slate-100 text-sm overflow-x-auto">
                            $$f(x) = \frac{L}{1 + e^{-k(x - x_0)}}$$
                        </div>

                            A classic <strong>Logistic Sigmoid Function</strong> modeling biological saturation.
                        </p>
                    </div>

                    <div>
                        <span class="text-xs font-bold text-slate-400 uppercase tracking-wider">Daily Weight Gain (DWG): \(f'(x)\)</span>
                        <div class="p-3 bg-emerald-50 rounded-xl mt-1 text-center font-mono border border-emerald-100 text-emerald-900 text-sm overflow-x-auto">
                            $$f'(x) = \frac{L \cdot k \cdot e^{-k(x - x_0)}}{\left(1 + e^{-k(x - x_0)}\right)^2}$$
                        </div>
                        <p class="text-xs text-slate-500 mt-1">
                            The derivative represents the **instantaneous rate of change** (growth rate in grams/day).
                        </p>

        const SCENARIOS = {
            closed: {
                L: 3100,      // Potential asymptotic maximum weight (g)

        let currentScenarioKey = 'closed';

        // Cumulative logistic growth function f(x)
        function f(x, params) {
            return params.L / (1 + Math.exp(-params.k * (x - params.x0)));
        }

        // Exact derivative f'(x) [Daily Weight Gain rate]
        function fPrime(x, params) {
            const expPart = Math.exp(-params.k * (x - params.x0));
            const denominator = 1 + expPart;
            return (params.L * params.k * expPart) / (denominator * denominator);
        }
